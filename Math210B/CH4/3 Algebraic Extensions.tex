Recall that $i$ is a root of the polynomial $x^2+1$ in $\Q[x]$, so we say that $i$ is algebraic. $\pi$, however, is not a root of any nonzer polynomial in $\Q[x]$ and so we say that $\pi$ is not algebraic over $\Q$. $\pi$ is a root of the polynomial $x - \pi$, so $\pi$ is algebraic over $\R$.

\begin{definition}[Algebraic, Transcendental] \leavevmode \\
    The element $\alpha \in K$ is algebraic over $F$ if $\alpha$ is a root of any nonzero polynomial in $F[x]$. If $\alpha$ is not algebraic over $F$, then $\alpha$ is said to be transcendental over $F$. The extension $K/F$ is algebraic if every element of $K$ is algebraic over $F$.
\end{definition}

\begin{proposition}
    Let $\alpha$ be algebraic over $F$. Then there is a unique, monic, irreducible polynomial $m_{\alpha, F} \in F[x]$ which has $\alpha$ as a root. A polynomial $f(x) \in F[x]$ has $\alpha$ as a root if and only if $m_{\alpha, F}(x)$ divides $f(x)$ in $F[x]$.
\end{proposition}

\begin{proof}
    Let $g(x) \in F[x]$ be a polynomial with minimal degree which has $\alpha$ as a root. Multiplying $g(x)$ by a constant will give a monic polynomial, so without loss of generality assume $g(x)$ is monic. Suppose $g(x)$ is irreducible in $F[x]$, say $g(x) = a(x)b(x)$ where $0 < \deg a(x), \deg b(x) < \deg g(x)$. Then
    $$g(\alpha) = a(\alpha)b(\alpha) = 0$$
    Since $K$ is a field and hence an integral domain, then $a(\alpha) = 0$ or $b(\alpha) = 0$, contradicting the minimality of $\deg g(x)$. It follows that $g(x)$ is a monic, irreducible polynomial in $F[x]$ with $\alpha$ as a root. Suppose now that $f(x) \in F[x]$ has $\alpha$ as a root. By the division algorithm 
    $$\exists q(x), r(x) \in F[x] \st f(x) = q(x)g(x) + r(x)$$
    where $r(x) = 0$ or $\deg r(x) < \deg g(x)$. Then
    \begin{align*}
        0 = f(\alpha) &= q(\alpha)g(\alpha) + r(\alpha) \\
        &= 0 + r(\alpha) \\
        &= r(\alpha)
    \end{align*}
    By the minimality of $\deg g(x)$, $r(x) = 0$ since $\alpha$ is a root of $r(x)$. hence $g(x)$ divides any polynomial with $\alpha$ as a root in $F[x]$. If $g(x) \mid h(x)$ where $h(x) \in F[x]$, then $\alpha$ is a root of $h(x)$.

    Since $g(x)$ divides any polynomial with $\alpha$ as a root, $g(x)$ divides monic polynomials with $\alpha$ as a root. That is, for any monic polynomial $p(x) \in F[x]$ with $\alpha$ as a root which has minimal degree
    $$c \cdot g(x) = p(x)$$
    so $g(x)$ is unique, and we let $g(x) = m_{\alpha, F}(x)$ from the statement.
    \qed
\end{proof}

\begin{definition}[Minimal Polynomial] \leavevmode \\
    The polynomial $m_{\alpha, F}(x)$ (or just $m_\alpha(x)$ if $F$ is understood) in the previous proposition is called the minimal polynomial of $\alpha$ over $F$. The degree of $m_\alpha (x)$ is called the degree of $\alpha$.
\end{definition}

\begin{proposition}
    Let $\alpha$ be algebraic over $F$ and let $F(\alpha)$ be the field generated by $\alpha$ over $F$. Then
    $$F(\alpha) \cong \frac{F[x]}{(m_{\alpha, F}(x))}$$
    and
    $$[F(\alpha) : F] = \deg (m_{\alpha, F}(x)) = \deg (\alpha)$$
\end{proposition}

\begin{example}
    \begin{enumerate}
        \item Let $\alpha = \sqrt{2}$ and $F = \Q$. Then the minimal polynomial of $\alpha$ over $\Q$ is $m_\alpha(x) = x^2 - 2$. Hence $[\Q(\sqrt{2}) : \Q] = \deg (m_\alpha(x)) = 2$.
        \item Let $\alpha = \sqrt[n]{2}$ and $F = \Q$. Then the minimal polynomial of $\alpha$ over $\Q$ is $m_\alpha(x) = x^n - 2$. Hence $[\Q(\sqrt[n]{2}) : \Q] = \deg (m_\alpha(x)) = n$.
    \end{enumerate}
\end{example}

\begin{example}
    Let $p(x) = x^3-3x-1$. This is a cubic so it has at least one real root. By the rational root theorem,
    $$
    \begin{rcases*}
        p(1) = -3 \\
        p(-1) = 1
    \end{rcases*} \implies \text{$p(x)$ is irreducible over $\mathbb{Q}$.}
    $$
    Let $\alpha$ denote the real root of $p(x)$. Then
    $$[\Q(\alpha) : \Q] = 3.$$
\end{example}

We can now deduce the degree of a simple extension through the degree of the minimal polynomial. The next result relates the degree of a simple extension to whether the element generating the extension is algebraic.

\begin{proposition}
    The element $\alpha$ is algebraic over $F$ if and only if the simple extension $F(\alpha)$ is finite. More precisely, if $\alpha$ is an elment of an extension of degree $n$ over $F$, then $\alpha$ satisfies a polynomial of degree at most $n$ over $F$ and if $\alpha$ satisfies a polynomial of degree $n$ over $F$, then the degree of $F(\alpha)$ over $F$ is at most $n$.
\end{proposition}

\begin{proof}
    If $\alpha$ is algebraic over $F$, then $\alpha$ satisfies a polynomila of positive degree, say $n$, over $F$. Thus
    $$[F(\alpha) : F] = \deg (m_{\alpha, F}(x)) \leq n$$
    since $m_{\alpha, F}(x)$ divides any polynomial satisfied by $\alpha$. \\
    Conversely, suppose $\alpha$ is an element of an extension of degree $n$ over $F$. Then the $n+1$ elements $1, \alpha, \alpha^2, ..., \alpha^n$ of $F(\alpha)$ are linearly independent over $F$. That is, we have
    $$b_0 + b_1 \alpha + b_2 \alpha^2 + ... + b_n \alpha^n = 0$$
    for some $b_i \in F$ with not all zero. Hence $\alpha$ is a root of
    $$p(x) = b_0 + b_1 x + b_2 x^2 + ... + b_n x^n \in F[x]$$
    where $p(x) = 0$ and $\deg(p(x)) \leq n$. Hence $\alpha$ is algebraic over $F$.
    \qed
\end{proof}

\begin{corollary}
    If the extension $K/F$ is finite, then it is algebraic.
\end{corollary}

\begin{proof}
    If $\alpha \in K,$ then the subfield $F(\alpha)$ is a subspace of the vector space $K$ over $F$. Hence, $[F(\alpha) : F] \leq [K : F]$. Since $[K : F] \in \N$, $F(\alpha)$ is a finite extension. Thus $\alpha$ is algebraic.
    \qed
\end{proof}

\begin{theorem} [Tower Law] \leavevmode \\
    Let $F \subseteq K \subseteq L$ be fields. Then
    $$[L : F] = [L : K][K:F]$$
\end{theorem}

\begin{proof}
    Suppose $[L : K] = m$ and $[K : F] = n$ for $m,n \in \N$. Let $\alpha_1, ..., \alpha_m$ be a basis for $L$ over $K$ and let $\beta_1, ..., \beta_n$ be a basis for $K$ over $F$. Then every element of $L$ can be written as 
    $$a_1\alpha_1 + ... + a_m\alpha_m \text{ for } a_i\in K$$
    Since $a_i$ are elements of $K$ we have
    $$a_i = b_{i1}\beta_1 + b_{i2}\beta_2 + ... + b_{in}\beta_n \text{ for } b_{ij} \in F$$
    Our elements can be written
    $$\left(\sum_{j=1}^{n}b_{1j}\beta_1\right)\alpha_1 + \left(\sum_{j=1}^{n}b_{2j}\beta_2\right)\alpha_2+...+\left(\sum_{j=1}^{n}b_{mj}\beta_m\right)\alpha_m = \sum_{i=1}^{m}\sum_{j=1}^{n}b_{ij}\beta_j\alpha_i$$
    where there are $m\cdot n$ elements $\beta_j\alpha_i$ and $b_{ij}$ are in $F$. Hence, $\beta_j\alpha_i$ span $L$ as a vector space over $F$.
    Suppose now that
    $$\sum_{i=1}^{m}\sum_{j=1}^{n}b_{ij}\beta_j\alpha_i = 0 \text{ for } b_{ij} \in F$$
    Then since the $\beta_j$ form a basis for $K$ over $F$, the equation above can be written as
    $$a_1\alpha_1 + ... + a_m \alpha_m = 0$$
    where $a_i \in K$ since
    $$\left(\sum_{j=1}^{n}b_{ij}\beta_j\right) \in K$$
    Since the $\alpha_i's$ form a basis for $L$ over $K$, it follows that $a_i = 0$ for all $i$. Thus
    $$b_{i1}\beta_1 + ... + b_{in}\beta_n = 0 \text{ for } 1 \leq i \leq m$$
    Since the $\beta_j's$ form a basis of $K$ over $F$, this implies that $b_{ij} = 0$ for all $i,j$. Hence $\alpha_i\beta_j$ are all linearly independent over $F$ and hence form a basis for $L$ over $F$. That is,
    $$[L : F] = [L : K][K : F].$$
    The cases where one of the extensions is infinite is in the book.
    \qed
\end{proof}

\begin{corollary}
    Suppose $L/F$ is a finite extension and let $K$ be a subfield of $L$ containing $F$, $F\subseteq K \subseteq L$. Then 
    $$[K : F] \mid [L : F]$$
\end{corollary}

\begin{example}
    Let $\alpha = 1 + \sqrt[3]{2} + \sqrt[3]{4}$. What is the degree of the extension $\Q(\alpha)?$
    $$\alpha = 1 + \sqrt[3]{2} + \left(\sqrt[2]{2}\right)^2 \in \Q(\sqrt[3]{2})$$
    Since $\alpha \in \Q(\sqrt[3]{2})$, $\Q(\alpha) \subseteq \Q(\sqrt[3]{2})$. Thus
    $$3 = [\Q(\sqrt[3]{2}):\Q] = [\Q(\sqrt[3]{2}) : \Q(\alpha)][\Q(\alpha) : \Q]$$
    Consider $[\Q(\sqrt[3]{2}):\Q(\alpha)]=3$ and $[\Q(\alpha) : \Q] = 1$. Then
    $$[\Q(\alpha) : \Q] = 1 \implies \Q(\alpha) = \Q$$
    However, $\alpha \not \in \Q$. Then it must be the case that $[\Q(\sqrt[3]{2}) : \Q(\alpha)] = 1$ and $[\Q(\alpha) : \Q] = 3.$ Hence
    $$\Q(\sqrt[3]{2}) = \Q(\alpha)$$
\end{example}

\begin{example}
    Let $\alpha$ be the real root of $x^3-3x-1.$ We will show $\sqrt{2} \not \in \Q(\alpha)$. That is, we will show $\sqrt{2} \neq a+b\alpha + c\alpha^2$ for any $a,b,c \in \Q$.
    For the sake of contradiction, suppose $\sqrt{2} \in \Q(\alpha)$. Then
    $$\Q(\sqrt{2})\subseteq \Q(\alpha) \implies [\Q(\alpha) : \Q] = [\Q(\alpha) : \Q(\sqrt{2})][\Q(\sqrt{2}): \Q] = [\Q(\alpha) : \Q(\sqrt{2})] \cdot 2.$$
    Hilariously, $2\mid 3$, a contradiction.
\end{example}

\begin{example}
    $\sqrt[6]{2}$ is the positive sixth root of $2$. Since
    $$m_{\sqrt[6]{2}, \Q}(x) = x^6-2$$
    we have $[\Q(\sqrt[6]{2}) : \Q]=6$. Consider $\left(\sqrt[6]{2}\right)^3=\sqrt{2}\in \Q\left(\sqrt[6]{2}\right).$ Thus
    \begin{align*}
        &\Q\left(\sqrt{2}\right) \subseteq \Q\left(\sqrt[6]{2}\right) \\
        \implies &6=\left[\Q\left(\sqrt[6]{2}\right):\Q\right] = \left[\Q\left(\sqrt[6]{2}\right) : \Q\left(\sqrt{2}\right)\right]\left[\Q\left(\sqrt{2}\right) : \Q\right] = 2 \\
        \implies &\left[\Q\left(\sqrt[6]{2}\right):\Q\left(\sqrt{2}\right)\right] = 3
    \end{align*}
\end{example}

\begin{example}
    Let $\alpha = \sqrt[5]{2} + 1$. Find the degree of $\Q(\alpha)$.

    Notice
    \begin{align*}
        \alpha \in \Q(\sqrt[5]{2}) &= \set{a+b\sqrt[5]{2}+c\sqrt[5]{4}+d\sqrt[5]{8}+e\sqrt[5]{16} : a,b,c,d,e \in \Q} \\
        &\implies \Q(\alpha) \subseteq \Q(\sqrt[5]{2})
    \end{align*}
    Also,
    $$\left[\Q(\sqrt[5]{2}):\Q\right] = 5$$
    since $m_{\alpha, \Q}(x)=x^5-2$. Since $\Q(\alpha)$ is a field, $\alpha - 1 \in \Q(\alpha)$. That is,
    $$\sqrt[5]{2}\in \Q(\alpha) \implies \Q(\sqrt[5]{2}) \subseteq \Q(\alpha)$$
    Thus $\Q(\alpha) = \Q(\sqrt[5]{2})$, so the degree of $\Q(\alpha) = 5$.
\end{example}

\begin{lemma}
    $F(\alpha, \beta) = \left(F(\alpha)\right)(\beta)$. I.e., the field generated by $\alpha$ and $\beta$ over $F$ is the field generated by $\beta$ over $F(\alpha)$.
\end{lemma}

\begin{proof}
    Note that $F\subseteq F(\alpha, \beta), \alpha \in F(\alpha, \beta),$ and $\beta \in F(\alpha, \beta)$. Hence
    $$F(\alpha) \subseteq F(\alpha, \beta) \text{ and } \beta \in F(\alpha, \beta)$$
    By definition,
    $$\left(F(\alpha)\right)(\beta) \subseteq F(\alpha, \beta).$$
    Similarly,
    $$F(\alpha) \subseteq \left(F(\alpha)\right)(\beta) \text{ and } \beta \in \left(F(\alpha)\right)(\beta).$$
    Futhermore,
    $$F\subseteq \left(F(\alpha)\right)(\beta), \alpha \in \left(F(\alpha)\right)(\beta \text{ and } \beta \in \left(F(\alpha)\right))(\beta).$$
    We can conclude
    $$F(\alpha, \beta) \subseteq \left(F(\alpha)\right)(\beta)$$
    by definition.
    \qed
\end{proof}

\begin{example}
    $\left[\Q\left(\sqrt{2}, \sqrt{3}\right): \Q\right]=\left[\Q\left(\sqrt{2}, \sqrt{3}\right): \Q(\sqrt{3})\right]\left[\Q\left(\sqrt{3}\right): \Q\right]$
\end{example}

In general, we can extend this inductively. Let
$$K = F(\alpha_1, \alpha_2, ..., \alpha_k) = \left(F(\alpha_1, ..., \alpha_{k-1})\right)(\alpha_k) = F(\alpha_1,...,\alpha_{k-2})(\alpha_{k-1})(\alpha_k)$$
which gives a sequence of fields:
$$F \subseteq F_0 \subseteq F_1 \subseteq ... \subseteq F_k = K$$
where $F_{i+1} = F_i(\alpha_{i+1})$ for $i = 0,...,k-1$. If $\alpha_1, ..., \alpha_k$ are algebraic over $F$ with degrees $n_1, ..., n_k$ respectively, then
$$\left[F_{i+1}:F_i\right]=\left[F_i(\alpha_{i+1}):F_i\right] \leq \left[F_i(\alpha_{i+1}):F\right]=n_{i+1}$$
with equality when $m_{\alpha_{i+1}, F}(x)$ remains irreducible in $F_i[x].$ By the tower law,
$$\left[K : F\right] = \left[F_k : F_{k-1}\right]\left[F_{k-1}:F_{k-2}\right]\cdots \left[F_1:F\right] \leq n_1\cdot n_2 \cdots n_k.$$
We are able to use this to calculate basis for extensions $F(\alpha_1, ..., \alpha_k)$. We only need to examine this in the case $k=2$:
$$F(\alpha, \beta) \text{ where $\alpha, \beta$ are algebraic.}$$
Consider $F(\alpha, \beta) = \left(F(\alpha)\right)(\beta).$ We see the elements of $F(\alpha, \beta)$ can be written as
$$b_0 + b_1\beta + b_2 \beta^2 + ... + \beta_{d-1}\beta^{d-1}$$
where $d=\left[F(\alpha, \beta) : F(\alpha)\right]$ and $b_j \in F(\alpha)$. Since $b_j \in F(\alpha)$, we have
$$b_j = a_{0j} + a_{1j}\alpha + a_{2j}\alpha^2 + ... + a_{(n-1)j}\alpha^{n-1}$$
where $a_{ij} \in F$ and $n = \left[F(\alpha) : F\right].$ Hence the elements of $F(\alpha, \beta)$ can be written in the form
$$\sum_{\mathclap{\substack{0 \leq i \leq n-1 \\ 0 \leq j \leq d -1}}}a_{ij}\alpha^i \beta^j$$

\begin{note}
    The elements $\alpha^i\beta^j$ are an $F$ basis for $F(\alpha, \beta)$.
\end{note}

\begin{example}
    $\Q(\sqrt{2}, \sqrt{3})$.

    From the tower law, we know
    $$\left[\Q(\sqrt{2}, \sqrt{3}) : \Q\right] = \left[\Q(\sqrt{2}, \sqrt{3}) : \Q(\sqrt{2})\right]\left[\Q(\sqrt{2}) : \Q\right] \leq 4.$$
    This gives us two cases: $\left[\Q(\sqrt{2}, \sqrt{3}) : \Q(\sqrt{2})\right] \in \set{1, 2}$.

    Consider $\left[\Q(\sqrt{2}, \sqrt{3}) : \Q\sqrt{2}\right] = 1$. Then $\left[\Q(\sqrt{2}, \sqrt{3}) : \Q\right]=2 \implies \Q(\sqrt{2}, \sqrt{3}) = \Q(\sqrt{2})$ so $\sqrt{3}\in \Q(\sqrt{2})$. That is,
    $$\sqrt{3} = a+b \sqrt{2} \text{ for some } a, b \in \Q$$
    If $a = 0$ and $b \neq 0$, then
    \begin{align*}
        \sqrt{3} &=b\sqrt{2} \\
        \implies \sqrt{6} &= 2b \\
        \implies \sqrt{6}&\in \Q, \text{ a contradiction.}
    \end{align*}
    If $b = 0$ and $a \neq 0$, then $\sqrt{3} = 2a \implies \sqrt{3}\in \Q,$ a contradiction.

    Now, if $a,b \neq 0$ then
    \begin{align*}
        3 = \left(a+b\sqrt{2}\right)^2 &= a^2+2b^2+2\sqrt{2}ab \\
        \implies \dfrac{3-a^2-2b^2}{2ab}&=\sqrt{2} \\
        \implies \sqrt{2} &\in \Q, \text{ a contradiction.}
    \end{align*}
    Thus $\sqrt{3} \neq \Q(\sqrt{2}) \implies \left[\Q(\sqrt{2},\sqrt{3}) : \Q(\sqrt{2})\right] = 2.$

    What's more: $x^2 - 3$ is irreducible over $\Q(\sqrt{2})$. Furthermore,
    $$\left[\Q(\sqrt{2}, \sqrt{3}) : \Q\right]= 4$$
    and the basis elements are $\sqrt{3}^i \sqrt{2}^j$ for $i,j = 0,1$. That is,
    $$\Q(\sqrt{2}, \sqrt{3}) = \set{a+b\sqrt{2}+c\sqrt{3}+d\sqrt{6} : a,b,c,d \in \Q}$$
\end{example}
